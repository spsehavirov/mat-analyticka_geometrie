  \begin{frame}{K čemu to všechno?}
    Analytická geometrie (AnaGeo) je jedním ze způsobů, jak nahlížet na
    geometrické problémy (např. kde se protnou přímky, protnou se vůbec? Co je
    to přímka?)
	
	\medskip
    AnaGeo používá nástroje analýzy (rovnice, vlastnosti rovnic a funkcí) - je třeba
    mít na paměti, co rovnice a čísla reprezentují, protože se na první pohled bude
    zdát, že jsme se vrátili do 1. ročníku 
	
	\medskip
    Nicméně vše zasadíme do \textbf{vektorového prostoru} 
	
	\medskip
    \textbf{Příklad:} Bod ležící na přímce
  \end{frame}

  \begin{frame}{Pojmy}
    K tomu, abychom mohli bezpečně pokračovat, je třeba si udělat společnou slovní
    zásobu (kterou budu aktivně využívat):

      \[

      \]

      Vektor označujeme malým psacím písmenem s šipkou a jeho "směr" značíme do kulatých
      závorek:
      \[
        \vec{v}(2;-3)
      \]
    \end{alertblock}
  \end{frame}

	  \begin{frame}{Parametrické vyjádření}
	    \begin{alertblock}{Parametrické vyjádření přímky}
	      Nechť je přímka $p$ dána bodem $A[a_{x};a_{y}]$ a vektorem $\vec{v}(v_{x};v
	      _{y})$. Parametrickým vyjádřením přímky rozumíme všechny rovnice, které dokážou
	      vyjádřit všechny body $P$, pro které platí $P \in p$:
	      \[
	        P=A+t\cdot \vec{v}\ \ \ \ t\in \mathbb{R}
	      \]
	      Někdy se uvádí dvě rovnice pro výpočet příslušných souřadnic bodu $P[x,y]$:
	      \[
	        x=a_{x}+t\cdot v_{x} \ \ \ \ \ \ \ \ \ \
	      \]
	      \[
	        y=a_{y}+t\cdot v_{y} \ \ \ \ t\in \mathbb{R}
	      \]
	    \end{alertblock}
	  \end{frame}

  \section{Obecná rovnice přímky}
	  \begin{frame}{Motivace}
	    Někdy je nepohodlné psát dvě rovnice pro jednu přímku (případně rovnice, kde jsou matice...). Proto můžeme zavést jednoduší tvar pomocí toho, že se zbavíme parametru: 
	    $$x=2-t$$ $$\underline{y=1+2t}$$ $$2x+y=5$$
	    Tím jsme se sice zbavili problému rovnic, nicméně ztrácíme možnost "intuitivních" informací - nevíme směrový vektor a bod, který by na přímce ležel.
	  \end{frame}
	  
	 \begin{frame}{Obecná rovnice přímky}
	 	\begin{alertblock}{Obecná rovnice přímky}
	 			Obecnou rovnicí přímky rozumíme \textbf{jednu} rovnici ve tvaru: $$ax+by+c=0$$ Body $[x,y]$, které splňují rovnost, pak leží na přímce dané touto rovnicí.
	 	\end{alertblock}
	 	Tato rovnice má několik výhod - jednak je pouze jedna a také se v ní snadno ověřují body. \\
	 	Příklad: Ověřte, že bod $A[2,-1]$ leží na přímce $p: x-2y+1=0$ \\[2ex]
	 	Dosadíme bod do rovnice a ověříme: $2-2\cdot (-1)+1 = 0$ Vyšla nepravda - Bod na přímce neleží
	 \end{frame}
	 
	 \begin{frame}{Normálový vektor}
	 	Nicméně ztrácíme o přímce několik intuitivních informací - její směr a bod, který na ní leží. Ovšem máme alespoň něco...
	 	\begin{alertblock}{Normálový vektor}
	 	Normálový vektor přímky je takový vektor, který je na přímku \textbf{kolmý}. Platí, že pro přímku $p: ax+by+c=0$ je normálový vektor $$\vec{n_p}(a,b)$$
	 	\end{alertblock}
	 	A co $c$? To nám o přímce moc neřekne... Ve stručnosti - jak moc je přímky vzdálená od počátku. Neplatí však, že, když $c=5$ pak vzdálenost přímky od počátku je $5$ (nicméně v některých speciálních případech může)
	 \end{frame}
	 
	 \begin{frame}{Jak získat obecný tvar přímky?}
	  \begin{center} Potřebujeme něco ve tvaru $ax+by+c=0$ \end{center} 
	 	\begin{enumerate}[i) ]
	 	 \item Nejprve zjistím $a$ a $b$ pomocí \textit{normálového vektoru}
	 	 \item Dosadím za $x$ a $y$ souřadnice bodu, který na té přímce leží
	 	 \item Mám rovnici o jedné neznáme $c$ - zjistím $c$
	 	 \item Zapíšu místo čísel $a,b,c$ zjištěná čísla. 
	 	\end{enumerate}
	  Proces je sice trochu komplikovanější, ale za to je to jenom jedna rovnice, která má několik výhod... \\[2ex]
	  Například ověřování toho, zdali bod leží na přímce.
	 \end{frame}

	\begin{frame}{Výhody obecné rovnice přímky}
	Obecná rovnice přímky má mnoho výhod:
	 \begin{itemize}[- ]
	  \item Je pouze jedna
	  \item Může se snadno upravovat (použijeme ekvivalentní úpravy)
	  \item Ověření bodů je hračka
	  \item Řešení vzájemné polohy je o trošku jednoduší
	 \end{itemize}
	Ale proč bychom chtěli tento tvar upravovat? Třeba do tvaru, který dobře známe:
	$$ax+by+c=0$$ $$\dfrac{a}{b}x+y+\dfrac{c}{b}=0$$ $$y=-\dfrac{a}{b}x-\dfrac{c}{b}$$ 
	\end{frame}
	
	\begin{frame}{Směrnicový tvar přímky}
	 \begin{alertblock}{Směrnicový tvar přímky}
	  \begin{minipage}{0.63\textwidth}
	   Směrnicovým tvarem přímky rozumíme následující tvar přímky: $$y=kx+q$$ kde $k$ je směrnicí přímky a $q$ je $y$-ová souřadnice průsečíku na ose $y$
	  \end{minipage}  \hspace{0.3cm}
	  \begin{minipage}{0.3\textwidth}
	   \begin{tikzpicture}[scale=0.6]
	    \draw[color=gray] (-1.2,-1.2) grid (5.2,4.2);
		\draw[->,thick] (-1.4,0)--(5.4,0) node[above]{$x$};
		\draw[->,thick] (0,-1.4)--(0,4.4) node[right]{$y$};
		\draw[BrickRed,thick] (1,1)-- node[pos=0.5,below]{$\Delta x$} (4,1);
		\draw[NavyBlue, thick] (4,1)-- node[pos=0.5,right]{$\Delta y$} (4,3);
		\draw[thick](-2,-1)--(5.5,4) node[below]{$p$};
		\draw[purple, ->] (1,1.2)-- node[pos=0.5, sloped, above]{$\vec{s_p}$}(4,3.2);
	   \end{tikzpicture}
	  \end{minipage}
	 \end{alertblock}
	 Jestliže přímka $p$ má směrový vektor $\vec{s_p}(s_x,s_y)$ pak platí: $$k=\dfrac{\textcolor{NavyBlue}{\Delta y}}{\textcolor{BrickRed}{\Delta x}} = \dfrac{s_y}{s_x}$$
	\end{frame}
